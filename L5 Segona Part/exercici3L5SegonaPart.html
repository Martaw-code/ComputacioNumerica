function r = trap(f,a,b,granularitat)
%Implementaci√≥ escollida per implementar l'apartat de trapezis
suma = 0;
it = a+granularitat;
while it < b
suma = suma + f(it);
it = it + granularitat;
end
r = granularitat/2 * (f(a) + 2*suma + f(b));
end
 
 
%Implementaci√≥ escollida per implementar l'apartat de simpson
function r = simp(f,a,b,granularitat)
x = a:granularitat:b;
r = 0;
n = length(x);
mid = (x(1:n-1) + x(2:n))/ 2;
r = sum(granularitat/6 * (f(x(1:n-1)) + 4*f(mid) + f(x(2:n))));
end
 
 
%Implementaci√≥ escollida per implementar l'apartat de romberg
function [R] = rombergM(f,a,b,n,tol)
% Entrada - integral d'f
% - a i b s√≥n l√≠mits superior i inferior d'integraci√≥
% - n √©s el nombre m√†xim de files de la taula
% - tol √©s la toler√†ncia
 
% Sortida - R √©s la taula de Romberg
% - err √©s l'estimaci√≥ de l'error (optatiu)
% - h √©s la granuaritat utilitzada (optatiu)
M = 1;
h = b-a;
err = 1;
J = 0;
R = zeros(8,8); %se pot canviar la mida R(n,n)
R(1,1) = h * (feval(f,a) + feval(f,b))/2;
while((err>tol) && (J<n)) || (J<8) %podem canviar la mida, variant (J<N)
J = J+1;
h = h/2;
s = 0;
for p=1:M
x = a+h*(2*p-1);
s = s+feval(f,x);
end
R(J+1,1) = R(J,1)/2+h*s;
M = 2 * M;
for K=1:J
R(J+1,K+1) = R(J+1,K) + (R(J+1,K) - R(J,K))/(4^K-1);
end
err = abs(R(J,J) - R(J+1,J+1));
end
%quad = R(J+1,J+1);
end
